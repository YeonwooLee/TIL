```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;

public class Main {
	public static void main(String[] args) throws IOException{
		BufferedReader br= new BufferedReader(new InputStreamReader(System.in));
		int n = Integer.parseInt(br.readLine()); //tc
		
		for(int i=0;i<n;i++) {
			int matchesCount = Integer.parseInt(br.readLine());//성냥 수
			long[] minDp = new long[101];//minDp[i] = i개 썼을 때 최소값
			
			//숫자 한개를 표현하는데 사용하는 개수는 2~7개 사이
			Arrays.fill(minDp, Long.MAX_VALUE);// 왜 하는지?? 나중에 다익스트라마냥 비교하나
			minDp[2] = 1;
			minDp[3] = 7;
			minDp[4] = 4;
			minDp[5] = 2;
			minDp[6] = 6;//6개최소 0인데 맨 앞에 못오니까 6
			minDp[7] = 8;
			minDp[8] = 10;//8은 왜해주냐?
			
			String[] add = {"1","7","4","2","0","8"};//더해지는 minDp
			for(int j=9;j<=100;j++) {//dp9~dp100 구하기
				for(int k = 2; k <= 7; k++) {//바텀업
					String line = "" + minDp[j-k]+ add[k-2];//아 이거 add[0]이 사실은 2짜리임
					minDp[j] = Math.min(Long.parseLong(line), minDp[j]);
				}
			}
			
			//큰 수만들기
			StringBuilder max = new StringBuilder();
			long a = matchesCount / 2;
			long b = matchesCount % 2;
			
			if(b==1) {//홀수
				max.append("7");
			}else {//짝수
				max.append("1");
			}
			
			for(int j=1;j<a;j++) {
				max.append("1");
			}
			System.out.println(minDp[matchesCount] + " " + max.toString());
		}
		
	}
	

}
```

``` kotlin
//package daily.y_2023.m_02.d_08.bj_성냥개비

import java.io.BufferedReader
import java.io.InputStreamReader;
import kotlin.math.min

fun main(){
    val br = BufferedReader(InputStreamReader(System.`in`))
    val T = br.readLine().toInt()

    for(tc in 0 until T){//T번 반복
        val n = br.readLine().toInt()

        //minDp
        val minDp = Array<Long>(101){Long.MAX_VALUE}
        minDp[2] = 1
        minDp[3] = 7
        minDp[4] = 4
        minDp[5] = 2
        minDp[6] = 6
        minDp[7] = 8
        minDp[8] = 10

        val add = arrayOf("1","7","4","2","0","8")
        for(i in 9..100){
            for(b in 0..5){
                minDp[i] = min((minDp[i-2-b].toString()+add[b]).toLong(),minDp[i])
            }
        }

        val mok = n/2
        val mod = n%2

        //6 -- 111
        //7 -- 711
        var max=""
        if(mod==1)max+="7"
        else max+="1"
//        println("mok"+mok.toString())
        max+="1".repeat(mok-1)

        println(minDp[n].toString()+" "+max)

    }
}
```
